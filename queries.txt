// ✅ 3. Afficher les contacts avec age > 18
////////////////////////////////////////////////////////////
db.contactlist.find({ age: { $gt: 18 } })

////////////////////////////////////////////////////////////
// ✅ 4. Afficher les contacts avec age > 18
// et prénom contenant "ah" (insensible à la casse)
////////////////////////////////////////////////////////////
db.contactlist.find({
  age: { $gt: 18 },
  FirstName: { $regex: "ah", $options: "i" }
})

////////////////////////////////////////////////////////////
// ✅ 5. Modifier le prénom de "Kefi Seif" en "Kefi Anis"
////////////////////////////////////////////////////////////
db.contactlist.updateOne(
  { LastName: "Kefi", FirstName: "Seif" },
  { $set: { FirstName: "Anis" } }
)

////////////////////////////////////////////////////////////
// ✅ 6. Supprimer tous les contacts avec age < 5
////////////////////////////////////////////////////////////
db.contactlist.deleteMany({ age: { $lt: 5 } })
